Defining views

 1 import os
 2 
 3 from django.conf import settings
 4 from django.shortcuts import redirect, render
 5 
 6 def comp_listing(request, directory_slug=None):
 7     """
 8 Output the list of HTML templates and subdirectories in the COMPS_DIR
 9 """
10     context = {}
11     working_dir = settings.COMPS_DIR
12     if directory_slug:
13         working_dir = os.path.join(working_dir, directory_slug)
14     dirnames = []
15     templates = []
16     items = os.listdir(working_dir)
17     templates = [x for x in items if os.path.splitext(x)[1] == '.html']
18     dirnames = [x for x in items if \
19                     not os.path.isfile(os.path.join(working_dir, x))]
20     templates.sort()
21     dirnames.sort()
22     context['directories'] = dirnames
23     context['templates'] = templates
24     context['subdirectory'] = directory_slug
25     return render(request, "comps/comp_listing.html", context)

Fabric

What is it?

  • A tool that lets you execute arbitrary Python functions via the command line
  • A library of subroutines (built on top of a lower-level library) to make executing shell commands over SSH easy and Pythonic.
